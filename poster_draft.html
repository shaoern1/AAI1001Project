0.3.3 Handling missing values

# Calculate mean for each year if at least one score is available, else set to NA
df_filtered <- df_filtered %>%
  group_by(country) %>%
  mutate(mean_score = ifelse(any(!is.na(score)), mean(score, na.rm = TRUE), NA)) %>%
  ungroup()

# Replace missing scores with the calculated mean for that country
df_filtered <- df_filtered %>%
  mutate(score = ifelse(is.na(score), mean_score, score)) %>%
  select(-mean_score)  # Remove the temporary mean_score column

# Display the resulting DataFrame
print(head(df_filtered))

# Identify the years present for each country in the filtered dataset
existing_data <- df_filtered %>% 
  distinct(country, year)

# Create a DataFrame for the missing countries with NA values for score, excluding those already in the dataset
missing_years <- expand.grid(country = missing_countries, year = c(2021, 2022, 2023))
missing_countries_df <- anti_join(missing_years, existing_data, by = c("country", "year"))
missing_countries_df$score <- NA

# Combine the original DataFrame with the missing countries DataFrame
df_combined <- bind_rows(df_filtered, missing_countries_df)
